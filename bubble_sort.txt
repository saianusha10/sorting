def bubblesort(array):
    n = len(array)
    for i in range(n):
        swap = False
        for j in range(0, n - i - 1):
            if array[j] > array[j + 1]:
                array[j], array[j + 1] = array[j + 1], array[j]
                swap = True
        if not swapped:
            break
    return array

a = bubblesort([23,12,20,50,60])
print(a)
